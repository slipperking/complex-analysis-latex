\subsubsection{The Factorization of Transcendental Meromorphic Functions}

\begin{theorem}
    Let \(f:\mathbb{C}\to\extcomplex\) be meromorphic of finite order \(\rho\) such that \(f(0)\neq 0,\infty\). If \(\qty{a_n}_{n=1}^\infty\) and \(\qty{b_n}_{n=1}^\infty\) are its zeros and poles in \(\mathbb{C}^*\) listed in order of nondecreasing moduli, counting multiplicities and zeros, then there exist integers \(p_1,p_2\leq\rho\) such that \[P_1(z)=\prod_{n=1}^\infty E_{p_1}\qty(\frac{z}{a_n}),\quad P_2(z)=\prod_{n=1}^\infty E_{p_2}\qty(\frac{z}{b_n})\] locally uniformly converge in \(\mathbb{C}\). Then \(f(z)=\ee^{\varphi(z)}\flatfrac{P_1(z)}{P_2(z)}\), where \(\varphi\) is a polynomial of degree \(\leq\rho\).
\end{theorem}
\begin{proof}
    Let \(f_1(z)=f(z)P_2(z)\), which is entire. Since \(N\qty(r,f_1)=0\leq N\qty(r,f)\) for any \(r>0\), it follows that
    \[T\qty(r,f_1)=m\qty(r,f_1)\leq m\qty(r,f)+m\qty(r,P_2)\]
    by \cref{itm:nevanlinnathreefunctionsmiscproperties_mprodsubadd} of \cref{prop:nevanlinnathreefunctionsmiscproperties}. By \cref{itm:nevanlinnacharacteristicproperties_prodsubadd} of \cref{prop:nevanlinnacharacteristicproperties}, we have \[T\qty(r,f_1)\leq T(r,f)+T\qty(r,P_2)\leq 2\max\qty{T(r,f),T\qty(r,P_2)}.\]
\end{proof}
