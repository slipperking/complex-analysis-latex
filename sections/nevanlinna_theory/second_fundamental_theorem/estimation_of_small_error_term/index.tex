\subsubsection{The Estimation of \texorpdfstring{\(S(r,f)\)}{the Small Error Term}}
The second part of the Second Fundamental Theorem is typically a suitable estimate on \(S(r,f)\).
\getkeytheorem{thm:nevanlinnasecondfundamentaltheorempart2}
Modern results vastly improve said estimations, and the search for a sharp estimation is one of the most challenging questions in Nevanlinna Theory.
\begin{lemma}\label{lem:nevanlinnasmallerrortermestimation1}
    Let \(z\in\mathbb{C}\) be arbitrary and fixed and suppose \(0<r<\infty\). Let \[E_k=\cbraces{\theta}{-\uppi<\theta<\uppi\wedge\abs{z-r\ee^{\ii\theta}}<kr}\] for each \(0<k\leq 1\). It follows that \[I=\int_{E_k}\log\abs{\frac{r}{z-r\ee^{\ii\theta}}}\dd{\theta}<\uppi k\qty(1-\log k).\]
\end{lemma}
\begin{proof}
    Let \(z=\rho\ee^{\ii\psi}\). Performing a substitution of \(\vartheta=\theta-\psi\), we have \[I=\int_{E'_k}\log\abs{\frac{r}{\rho-r\ee^{\ii\vartheta}}}\dd{\vartheta},\] where \[E'_k=\cbraces{\vartheta}{-\uppi-\psi<\vartheta<\uppi-\psi\wedge\abs{\rho-r\ee^{\ii\vartheta}}<kr}.\]
    Because the exponential is periodic every \(2\uppi\ii\), we may assume that \(E'_k\) is restricted to \(-\uppi<\vartheta<\uppi\) instead of \(-\uppi-\psi<\vartheta<\uppi-\psi\). The expression \(\abs{\rho-r\ee^{\ii\vartheta}}=kr\) gives at most two symmetric solutions \(\pm\vartheta_0\) (let \(\vartheta_0\geq0\)).

    Evidently, for \(\vartheta>\flatfrac{\uppi}2\) or \(\vartheta<-\flatfrac{\uppi}2\), \(\abs{\rho-r\ee^{\ii\vartheta}}\geq r\geq kr\), so it follows that \(\vartheta_0\leq\flatfrac{\uppi}2\), and furthermore, all \(-\vartheta_0<\vartheta<\vartheta_0\) lie in \(E'_k\) by the geometry of a circle. Since \[\abs{\rho-r\ee^{\ii\vartheta}}\geq\abs{\Im\qty(\rho-r\ee^{\ii\vartheta})}=r\abs{\sin\vartheta},\]
    assuming that \(E'_k\neq\emptyset\), it follows that
    \begin{align*}
        I&=2\int_0^{\vartheta_0}\log\abs{\frac{r}{\rho-r\ee^{\ii\vartheta}}}\dd{\vartheta}\leq2\int_0^{\vartheta_0}\log\csc\vartheta\dd{\vartheta}\leq2\int_0^{\vartheta_0}\log\tfrac{\uppi}{2\vartheta}\dd{\vartheta}\\
        &=2\vartheta_0\log\tfrac{\uppi}2-2\mathop{\mathmakebox[\widthof{\(\int\)}][l]{\int_0^{\vartheta_0}}}\log\vartheta\dd{\vartheta}=2\vartheta_0\log\tfrac{\uppi}2-2\qty(\vartheta_0\log\vartheta_0-\vartheta_0)=2\vartheta_0\qty(\log\tfrac{\uppi}{2\vartheta_0}+1).
    \end{align*}
    Since \(\qty(\flatfrac{\uppi}2)\sin x\geq x\) for \(x\in\qty[0,\flatfrac{\uppi}2]\), it follows that \[\vartheta_0\leq\tfrac{\uppi}2\sin\vartheta_0=\tfrac{\uppi k}2.\]
    Moreover, \(\vartheta\mapsto2\vartheta\qty(\log\tfrac{\uppi}{2\vartheta}+1)\) is an increasing function of \(0\leq\vartheta\leq\flatfrac{\uppi}2\); thus we may replace all instances of \(\vartheta_0\) with \(\flatfrac{\uppi k}2\), proving the lemma.
\end{proof}
\begin{lemma}\label{lem:nevanlinnasmallerrortermestimation2}
    Let \(\qty{z_\nu}_{\nu=1}^n\) be \(n\geq1\) (possibly indistinct) complex numbers. Define \(\textstyle\delta(z)=\min\cbraces{\abs{z-z_\nu}}{\nu\in\mathbb{N}_{\leq n}}\). It follows that for any \(r>0\), \[\frac1{2\uppi}\int_0^{2\uppi}\logp\frac{r}{\delta\qty(r\ee^{\ii\theta})}\dd{\theta}\leq2\log n+\tfrac12\]
\end{lemma}
\begin{proof}
    Define the set \[E_\nu=\cbraces{\theta\in(-\uppi,\uppi)}{\abs{z_\nu-r\ee^{\ii\theta}}<\tfrac{r}{n}}\]
    and let \(E=\textstyle\bigcup_{\nu=1}^n E_\nu\). For \(\theta\in E\), \(\delta\qty(r\ee^{\ii\theta})<\flatfrac{r}{n}\), implying that \(\tfrac{r}{\delta\qty(r\ee^{\ii\theta})}>n\). Define \[\log_0x=
        \begin{cases}
            \log x&x\geq n,\\
            0&\text{otherwise},
    \end{cases}(\geq 0)\]
    so that for \(\theta\) in the prescribed range, we have \[\logp\frac{r}{\delta\qty(r\ee^{\ii\theta})}=\log_0\frac{r}{\delta\qty(r\ee^{\ii\theta})}=\log_0\frac{r}{\min_{\nu\in\mathbb{N}_{\leq n}}\abs{r\ee^{\ii\theta}-z_\nu}}\leq\sum_{\nu=1}^n\log_0\frac{r}{\abs{r\ee^{\ii\theta}-z_\nu}}\]
    Since \(0<\flatfrac1n\leq1\), by virtue of \cref{lem:nevanlinnasmallerrortermestimation1} for \(k=\flatfrac1n\),
    \begin{multline}
        \frac1{2\uppi}\int_E\logp\frac{r}{\delta\qty(r\ee^{\ii\theta})}\dd{\theta}\leq\frac1{2\uppi}\sum_{\nu=1}^n\int_0^{2\uppi}\log_0\frac{r}{\abs{r\ee^{\ii\theta}-z_\nu}}\dd{\theta}\\
        =\frac1{2\uppi}\sum_{\nu=1}^n\int_0^{2\uppi}\log\frac{r}{\abs{r\ee^{\ii\theta}-z_\nu}}\dd{\theta}<\frac1{2\uppi}\sum_{\nu=1}^n\frac{\uppi}{n}\qty(1+\log n)=\tfrac12+\tfrac{\log n}2.\label{eq:nevanlinnasmallerrortermestimation2_Eset}
    \end{multline}
    For \(\theta\notin E\), for each \(\nu\) we have \(\abs{z_\nu-r\ee^{\ii\theta}}\geq\flatfrac{r}{n}\) and thus \(\delta\qty(r\ee^{\ii\theta})\geq\flatfrac{r}{n}\). It follows that
    \begin{equation}
        \frac1{2\uppi}\int_{\theta\notin E}\logp\frac{r}{\delta\qty(r\ee^{\ii\theta})}\dd{\theta}\leq\frac1{2\uppi}\int_0^{2\uppi}\logp n\dd{\theta}\leq\log n.\label{eq:nevanlinnasmallerrortermestimation2_Ecomplement}
    \end{equation}
    Combining \cref{eq:nevanlinnasmallerrortermestimation2_Eset,eq:nevanlinnasmallerrortermestimation2_Ecomplement} gives the desired result.
\end{proof}
\begin{proposition}\label{prop:nevanlinnalogdiffproximityestimate}
    If \(f\) is meromorphic on a neighborhood of \(\overline{D(0,R)}\), then for \(0<r<R\), letting \(c\) be the first nonzero coefficient of the Laurent of expansion of \(f\) about the origin,
    \begin{multline}
        m\qty(r,\tfrac{f'}f)<4\logp T(R,f)+4\logp\circ\logp\abs{\tfrac1c}\\
        +5\logp R+6\logp\tfrac1{R-r}+\logp\tfrac1r+14.\label{eq:nevanlinnalogdiffproximityestimate_statement}
    \end{multline}
\end{proposition}
\begin{proof}
    By the Poisson--Jensen Formula (\cref{thm:poissonjensenformula}), letting \(\rho=\tfrac12(R+r)\), for each non-singular point \(z\in\partial D(0,r)\),
    \begin{align*}
        \Re\log f(z)&=\frac1{2\uppi}\int_0^{2\uppi}\log\abs{f\qty(\rho\ee^{\ii\theta})}\Re\qty(\frac{\rho\ee^{\ii\theta}+z}{\rho\ee^{\ii\theta}-z})\dd{\theta}\\
        &\quad+\sum_{k=1}^{n(\rho,0,f)}\Re\log\qty(\frac{\rho\qty(z-a_k)}{\rho^2-\overline{a_k}z})-\sum_{j=1}^{n(\rho,f)}\Re\log\qty(\frac{\rho\qty(z-b_j)}{\rho^2-\overline{b_j}z}),
    \end{align*}
    where \(\qty{a_k}\) and \(\qty{b_j}\) are the respective zeros and poles. By \cref{eq:wirtingerderivativeofrealpartofholomorphicfunction}, we have \(\textstyle\pdv{z}(\Re g(z))=\tfrac12g'(z)\) for all holomorphic \(g\). Applying this with differentiation under the integral sign, we have that
    \begin{align*}
        \frac{f'(z)}{f(z)}&=\frac1{2\uppi}\int_0^{2\uppi}\log\abs{f\qty(\rho\ee^{\ii\theta})}\frac{2\rho\ee^{\ii\theta}\dd{\theta}}{\qty(\rho\ee^{\ii\theta}-z)^2}\\
        &\quad+\sum_{k=1}^{n(\rho,0,f)}\qty(\frac{\overline{a_k}}{\rho^2-\overline{a_k}z}-\frac1{a_k-z})+\sum_{j=1}^{n(\rho,f)}\qty(\frac1{b_j-z}-\frac{\overline{b_j}}{\rho^2-\overline{b_j}z}).
    \end{align*}
    Define \(\delta(z)=\min\qty(\cbraces{\abs{z-a_k}}{k\in\mathbb{N}_{\leq n(\rho,0,f)}}\cup\cbraces{\abs{z-b_j}}{j\in\mathbb{N}_{\leq n(\rho,f)}})\). Evidently the number of zeros and poles is given by \[n=n(\rho,f)+n(\rho,0,f).\]
    Since \[\abs{\rho^2-\overline{a_k}z}\geq\rho^2-r\abs{a_k}\geq\rho^2-r\rho,\qquad\abs{\rho^2-\overline{b_j}z}\geq\rho^2-r\rho,\]
    it follows that
    \begin{equation}
        \abs{\frac{\overline{a_k}}{\rho^2-\overline{a_k}z}}\leq\frac{\rho}{\rho^2-r\rho}=\frac1{\rho-r},\quad\abs{\frac{\overline{b_j}}{\rho^2-\overline{b_j}z}}\leq\frac{\rho}{\rho^2-r\rho}=\frac1{\rho-r},\label{eq:nevanlinnalogdiffproximityestimate_mobiusbounds}
    \end{equation} and additionally,
    \begin{equation}
        \abs{\frac1{b_j-z}}\leq\frac1{\delta(z)},\qquad\abs{\frac1{a_k-z}}\leq\frac1{\delta(z)}.\label{eq:nevanlinnalogdiffproximityestimate_inversionbounds}
    \end{equation}
    Furthermore,
    \begin{multline}
        \abs{\frac1{2\uppi}\int_0^{2\uppi}\log\abs{f\qty(\rho\ee^{\ii\theta})}\frac{2\rho\ee^{\ii\theta}\dd{\theta}}{\qty(\rho\ee^{\ii\theta}-z)^2}}\leq\frac1{2\uppi}\frac{2\rho}{\qty(\rho-r)^2}\int_0^{2\uppi}\abs{\log\abs{f\qty(\rho\ee^{\ii\theta})}}\dd{\theta}\\
        =\frac{2\rho}{\qty(\rho-r)^2}\qty[m\qty(\rho,f)+m\qty(\rho,\tfrac1f)]\label{eq:nevanlinnalogdiffproximityestimate_integralbounds}
    \end{multline}
    by \cref{itm:lognonnegativepartproperties_sumofreciprocallogs} of \cref{prop:lognonnegativepartproperties}. Combining \cref{eq:nevanlinnalogdiffproximityestimate_mobiusbounds,eq:nevanlinnalogdiffproximityestimate_inversionbounds,eq:nevanlinnalogdiffproximityestimate_integralbounds}, we have
    \begin{align*}
        \abs{\frac{f'(z)}{f(z)}}&\leq\frac{2\rho}{\qty(\rho-r)^2}\qty[m\qty(\rho,f)+m\qty(\rho,\tfrac1f)]+\sum_{k=1}^{n(\rho,0,f)}\qty(\frac1{\rho-r}+\frac1{\delta(z)})+\sum_{j=1}^{n(\rho,f)}\qty(\frac1{\rho-r}+\frac1{\delta(z)})\\
        &=\frac{2\rho}{\qty(\rho-r)^2}\qty[m\qty(\rho,f)+m\qty(\rho,\tfrac1f)]+n\qty(\frac{1}{\rho-r}+\frac{1}{\delta(z)}).
    \end{align*}
    By \cref{prop:nevanlinnafirsttheorematzero}, we have \(T(\rho,0,f)=T(\rho,f)-\log\abs{c}\), where \(c\) is the first nonzero coefficient of the Laurent expansion of \(f\) about the origin. It follows that \(m\qty(\rho,\tfrac1f)=T(\rho,f)-\log\abs{c}-N\qty(\rho,0,f)\) and \(m\qty(\rho,f)=T(\rho,f)-N\qty(\rho,f)\), thus \[m\qty(\rho,\tfrac1f)+m(\rho,f)=2T(\rho,f)-\log\abs{c}-N\qty(\rho,\tfrac1f)-N\qty(\rho,f)\leq 2T(\rho,f)+2\logp\abs{\tfrac1{c}},\]
    and
    \begin{align*}
        \abs{\frac{f'(z)}{f(z)}}&\leq\frac{4\rho}{\qty(\rho-r)^2}\qty[T(\rho,f)+\logp\abs{\frac1{c}}]+\frac{n}{r}\qty(\frac{r}{\rho-r}+\frac{r}{\delta(z)}).
    \end{align*}
    By the subadditive properties of \cref{prop:lognonnegativepartproperties}, we have
    \begin{align*}
        \logp\abs{\tfrac{f'(z)}{f(z)}}&\leq\logp\qty[\tfrac{4\rho}{\qty(\rho-r)^2}\qty[T(\rho,f)+\logp\abs{\tfrac1{c}}]]+\logp\qty[\tfrac{n}{r}\qty(\tfrac{r}{\rho-r}+\tfrac{r}{\delta(z)})]+\log 2\\
        &\leq\logp\rho+\logp\qty[\qty(\tfrac{2}{\rho-r})^2]+\logp\qty[T(\rho,f)+\logp\abs{\tfrac1{c}}]+\logp\tfrac{n}{r}\\
        &\quad+\logp\qty(\tfrac{r}{\rho-r}+\tfrac{r}{\delta(z)})+\log2\\
        &\leq\logp\rho+2\logp\tfrac{1}{\rho-r}+\logp T(\rho,f)+\logp\circ\logp\abs{\tfrac1{c}}+\logp\tfrac{n}{r}\\
        &\quad+\logp\tfrac{r}{\rho-r}+\logp\tfrac{r}{\delta(z)}+5\log2.
    \end{align*}
    Integrating on \(\abs{z}=r\) with \cref{lem:nevanlinnasmallerrortermestimation2} gives
    \begin{align}
        m\qty(r,\tfrac{f'}f)&\leq\logp\rho+2\logp\frac{1}{\rho-r}+\logp T(\rho,f)+\logp\circ\logp\abs{\tfrac1{c}}+\logp\tfrac{n}{r}\nonumber\\
        &\quad+\logp\frac{r}{\rho-r}+\frac1{2\uppi}\int_0^{2\uppi}\logp\frac{r}{\delta\qty(r\ee^{\ii\theta})}\dd{\theta}+5\log2\nonumber\\
        &\leq\logp\rho+3\logp\frac{1}{\rho-r}+\logp T(\rho,f)+\logp\circ\logp\abs{\frac1{c}}+3\logp n+\logp\tfrac{1}{r}\nonumber\\
        &\quad+\logp r+\tfrac12+5\log2.\label{eq:nevanlinnalogdiffproximityestimate_proximityprimaryestimate}
    \end{align}
    To derive a more useful estimate for \(n\), we note that
    \begin{gather*}
        N(R,f)\geq\int_\rho^R\frac{n(x,f)\ddx}{x}\geq\int_\rho^R\frac{n(\rho,f)\ddx}{R}=n(\rho,f)\frac{R-\rho}{R},\\
        N(R,0,f)\geq n(\rho,0,f)\frac{R-\rho}R.
    \end{gather*}
    Thus,
    \begin{align*}
        n&=n(\rho,f)+n(\rho,0,f)\leq\tfrac{R}{R-\rho}\qty[N(R,f)+N(R,0,f)]\\
        &\leq\tfrac{R}{R-\rho}\qty[T(R,f)+T(R,0,f)-m(R,f)-m(R,0,f)]\\
        &\leq\tfrac{R}{R-\rho}\qty[2T(R,f)+\log\abs{\tfrac1{c}}]\leq\tfrac{2R}{R-\rho}\qty[T(R,f)+\logp\abs{\tfrac1{c}}].
    \end{align*}
    Using the \(\logp{}\) subadditivity properties (as with earlier), we have
    \begin{align*}
        \logp{n}&\leq\logp{\frac{2R}{R-\rho}}+\logp T(R,f)+\logp\circ\logp\abs{\frac1c}+\log 2\\
        &\leq\logp R+\logp{\frac{1}{R-\rho}}+\logp T(R,f)+\logp\circ\logp\abs{\frac1c}+2\log 2
    \end{align*}
    Substituting this into \cref{eq:nevanlinnalogdiffproximityestimate_proximityprimaryestimate}, and using \(R-\rho=\rho-r=\tfrac{R-r}2\) we have that (recognizing \(\logp r,\logp\rho<\logp R\))
    \begin{multline*}
        \begin{aligned}
            m\qty(r,\frac{f'}f)&\leq\logp\rho+3\logp\frac{2}{R-r}+\logp T(\rho,f)+\logp\circ\logp\abs{\frac1{c}}+\logp\frac{1}{r}\nonumber\\
            &\qquad+\logp r+\tfrac12+5\log2+3\logp R+3\logp{\frac{2}{R-r}}\\
            &\qquad\qquad+3\logp T(R,f)+3\logp\circ\logp\abs{\frac1c}+6\log 2
        \end{aligned}\\
        \begin{aligned}
            {\!}&<5\logp R+17\log 2+6\logp\frac1{R-r}+4\logp T(\rho,f)+4\logp\circ\logp\abs{\frac1c}+\logp\frac1r+\frac12\\
            {\!}&\leq 5\logp R+17\log 2+6\logp\frac1{R-r}+4\logp T(R,f)+4\logp\circ\logp\abs{\frac1c}+\logp\frac1r+\frac12,
        \end{aligned}
    \end{multline*}
    since \(T(\rho,f)\leq T(R,f)\) (\cref{thm:nevanlinnacharacteristicnondecreasingconvex}). This proves the proposition.
\end{proof}

\begin{lemma}[\textsc{Borel}]\label{lem:nevanlinnaborel}
    Let \(T:\left[r_0,R_0\right)\to\mathbb{R}\) be an increasing, continuous function satisfying \(T>1\) on all of its domain (where \(r_0<R_0\leq\infty\)).
    \begin{enumerate}
        \item If \(R_0=\infty\), for \(r\geq r_0\) outside a set \(E_0\subset\mathbb{R}_{\geq r_0}\) whose linear measure does not exceed 2 (satisfying \(\textstyle\int_{E_0}\dd{r}\le2\) with respect to the Lebesgue measure\footnote{``Linear measure'' refers to length. Here, the condition is equivalent to the inequality given by \(\textstyle\inf\cbraces{\sum_k\abs{I_k}}{E\subseteq\bigcup_k I_k}\le2\) where \(\cbraces{I_k}\) are finite open intervals.}),
            \begin{equation}
                T\qty(r+\frac1{T(r)})<2T(r)\label{eq:nevanlinnaborel_infiniteradius}
            \end{equation}
        \item If \(R_0<\infty\), then \(\exists E_0\subset\left[r_0,R_0\right)\) satisfying \(\textstyle\int_{E_0}\frac{\dd{r}}{R_0-r}\leq2\) such that for \(r\in\left[r_0,R_0\right)\setminus E_0\),
            \begin{equation}
                T\qty(r+\frac{R_0-r}{\ee T(r)})<2T(r)\label{eq:nevanlinnaborel_finiteradius}
            \end{equation}
            Moreover, for any \(R,\rho,\rho'\in\qty(0,R_0)\) such that \(R-\rho'<\ee^{-2}\qty(R-\rho)\) (implying that \(\rho<\rho'\)), \(\exists r\in(\rho,\rho')\) such that \(r\notin E_0\).
    \end{enumerate}
\end{lemma}
\begin{proof}
    We observe the two cases separately.
    \begin{enumerate}
        \item Assume \(R_0=\infty\). Then define \(r'_0=r_0+\flatfrac1{T\qty(r_0)}\), and \[r_1=\inf\cbraces{r>r'_0}{\text{\cref{eq:nevanlinnaborel_infiniteradius} fails}},\quad r'_1=r_1+\flatfrac1{T\qty(r_1)}>r_1,\]
            and let \[r_2=\inf\cbraces{r>r'_1}{\text{\cref{eq:nevanlinnaborel_infiniteradius} fails}},\quad r'_2=r_2+\flatfrac1{T\qty(r_2)}>r_2,\] repeating this process recursively by \[r_n=\inf\cbraces{r>r'_{n-1}}{\text{\cref{eq:nevanlinnaborel_infiniteradius} fails}},\quad r'_n=r_n+\flatfrac1{T\qty(r_n)}>r_n.\] It follows that \(r_n\geq r'_{n-1}>r_{n-1}\), and \(r_n-r_{n-1}\geq r'_{n-1}-r_{n-1}\).

            Let \(N\in\mathbb{N}\cup\qty{\infty}\) be the largest \(n\) for which the infimum in \(r_n\) is defined. Observe that by the definition of the infimum, for each \(n\), \cref{eq:nevanlinnaborel_infiniteradius} must be satisfied throughout each \(\qty(r'_{n-1},r_n)\). Indeed, if there exists some value within this interval for which \cref{eq:nevanlinnaborel_infiniteradius} fails, then \(r_n\), which was defined by an infimum, would then be at most this new value, contradicting the assertion that it lay within the interval. It follows that the exceptional set must lie outside of the disjoint union; in other words, \[E_0\subseteq\mathbb{R}_{\geq r_0}\setminus\bigcup_{n=1}^N\qty(r'_{n-1},r_n)=\bigcup_{n=0}^N\qty[r_n,r'_n].\]
            We state without proof, that subadditivity holds:
            \[\int_{E_0}\dd{r}\leq\sum_{n=0}^N\qty(r'_n-r_n).\]
            By construction, for points arbitrarily close to \(r_n\) or at \(r_n\) itself, \[T\qty(r+\frac1{T(r)})\geq 2T(r),\]
            continuity allows (in either case) this inequality to be extended to \(r=r_n\). Then for each \(n\), \[T\qty(r'_n)=T\qty(r_n+\frac1{T\qty(r_n)})\geq 2T\qty(r_n),\quad T\qty(r'_n)\leq T\qty(r_{n+1})\implies T\qty(r_{n+1})\geq 2T\qty(r_n).\]
            Using induction, \(T(r_n)\geq 2^nT\qty(r_0)\), implying that \[r'_n-r_n=\frac1{T(r_n)}\leq\frac1{2^nT\qty(r_0)}\implies\sum_{n=0}^N\qty(r'_n-r_n)\leq\frac1{T(r_0)}\sum_{n=0}^\infty\frac1{2^n}=\frac2{T\qty(r_0)}\leq2.\]
        \item Suppose \(R_0<\infty\) in the theorem statement. Let \(T_0(\rho)=T\qty(R_0-\ee^{-\rho})\), where \(\rho_0=\log\tfrac1{R_0-r_0}\leq\rho<\infty\), so that \(T_0\) is an increasing function on \(\left[\rho_0,\infty\right)\). By the result above, \(\exists E_0'\subseteq\left[\rho_0,\infty\right)\) such that \[\int_{E_0'}\dd{\rho}\leq 2,\qquad T_0\qty(\rho+\frac1{T_0(\rho)})<2T_0(\rho)\quad\forall \rho\notin E'_0.\]
            Substitute \(\rho=\log\tfrac1{R_0-r}\), \(r=R_0-\ee^{-\rho}\), \(E_0=\qty(\rho\mapsto R_0-\ee^{\rho})\qty(E'_0)\), so that \(T_0(\rho)=T(r)\), and \[\int_{E_0}\frac{\dd{r}}{R_0-r}\leq 2,\qquad T\qty(R_0-(R_0-r)\ee^{-\frac1{T(r)}})<2T(r)\quad\forall r\notin E_0.\]
            We are left to bound the inequality to be in the form of \cref{eq:nevanlinnaborel_finiteradius}. For \(x\in(0,1)\), \(\ee^{-x}<1-\tfrac{\ee-1}\ee x<1-\flatfrac{x}{\ee}\), thus for \(r\) within \([r_0,R_0)\) outside \(E_0\), we have from \(\ee^{-\frac1{T(r)}}<1-\tfrac{1}{\ee T(r)}\). Hence, \[\textstyle T\qty(R_0-(R_0-r)\ee^{-\frac1{T(r)}})\geq T\qty(R_0-(R_0-r)\qty(1-\frac{1}{\ee T(r)}))=T\qty(r+\frac{R_0-r}{\ee T(r)}).\]
            If \(\rho,\rho'\) are chosen such that \(R_0-\rho'<\ee^{-2}\qty(R_0-\rho)\), then \[\int_{\rho}^{\rho'}\frac{\dd{r}}{R_0-r}=\log\qty(\frac{R_0-\rho}{R_0-\rho'})>\log\qty(\ee^2)=2.\]
            Since \(\textstyle\int_{E_0}\frac{\dd{r}}{r_0-r}\leq 2\), there then \((\rho,\rho')\not\subseteq E_0\). Thus, \(\exists r\in(\rho,\rho')\) such that \(r\notin E_0\).\qedhere
    \end{enumerate}
\end{proof}
\begin{theorem}[name=\textsc{Second Fundamental Theorem of Nevanlinna Theory, Part 2},store=thm:nevanlinnasecondfundamentaltheorempart2]\label{thm:nevanlinnasecondfundamentaltheorempart2}
    Let \(f:D\qty(0,R_0)\to\extcomplex\) (\(0<R_0\leq\infty\)) be meromorphic and non-constant. Define \(S(r,f)\) as in \cref{eq:nevanlinnasecondfundamentaltheorempart1_smallerrorterm}. Then
    \begin{enumerate}
        \item If \(R_0=\infty\) and \(f\) has finite order, then
            \begin{equation}
                S(r,f)=\order{\log T(r,f)}+\order{\log r}\label{eq:nevanlinnasecondfundamentaltheorempart2_intermediate_entirecomplexplane}
            \end{equation} as \(r\to\infty\) through all values.
        \item If \(R_0=\infty\) and \(f\) has infinite order, then \cref{eq:nevanlinnasecondfundamentaltheorempart2_intermediate_entirecomplexplane} as \(r\to\infty\) through all values of \(r\) outside a set \(E\) of finite linear measure.
        \item If \(R_0<\infty\), then
            \begin{equation}
                S(r,f)=\order{\logp T(r,f)+\log\frac1{R_0-r}}\label{eq:nevanlinnasecondfundamentaltheorempart2_intermediate_largedisk}
            \end{equation}
            as \(r\to R_0^-\) outside a set \(E\) satisfying
            \begin{equation}
                \int_E\frac{\dd{r}}{R_0-r}<\infty.
            \end{equation}
            Moreover, for any \(\rho,\rho'\in\qty(0,R_0)\) such that \(R_0-\rho'<\ee^{-2}\qty(R_0-\rho)\) (implying that \(\rho<\rho'\)), \(\exists r\in(\rho,\rho')\) such that \(r\notin E\).
    \end{enumerate}
\end{theorem}
\begin{proof}
    As a function of \(r\), \[S(r,f)=m\qty(r,\frac{f'}f)+m\qty(r,\frac{\phi'}{\phi})+\text{constant},\]
    where \(\textstyle\phi(z)=\prod_{\nu=1}^q\qty(f(z)-a_{\nu})\), where the constant term does not depend on \(r\).

    In all three perceivable cases,
    \begin{equation}
        T\qty(r,\phi)\leq\sum_{\nu=1}^q T\qty(r,f-a_\nu)=qT\qty(r,f)+\order{1},\qquad\logp T(r,\phi)=\logp T(r,f)+\order{1}\label{eq:nevanlinnasecondfundamentaltheorempart2_intermediate}
    \end{equation}
    by \cref{prop:nevanlinnacharacteristicproperties,thm:nevanlinnafirstfundamentaltheorem}. In the first case, where \(R_0=\infty\), by \cref{prop:nevanlinnalogdiffproximityestimate} with \(R=2r\),
    \begin{gather*}
        m\qty(r,\frac{f'}f)<4\logp T(R,f)+\order{1}+5\logp\qty(2r)=\order{\logp T(2r,f)}+\order{\logp r}.\\
        m\qty(r,\frac{\phi'}\phi)<4\logp T(R,\phi)+\order{1}+5\logp\qty(2r)=\order{\logp T(2r,\phi)}+\order{\logp r}.
    \end{gather*}
    Since \(f\) has finite order, \(T(2r,f)=\order{(2r)^{\rho+\varepsilon}}=\order{r^{\varrho}}\), so \(\logp T(R,f)=\order{\logp r}\), thus as \(r\to\infty\), using \cref{eq:nevanlinnasecondfundamentaltheorempart2_intermediate} we have \[m\qty(r,\frac{f'}f)=\order{\log r},\qquad m\qty(r,\frac{\phi'}\phi)=\order{\logp T(2r,\phi)}+\order{\logp r}=\order{\log r}.\] Then \(S(r,f)=\order{\log r}\), clearly satisfying \cref{eq:nevanlinnasecondfundamentaltheorempart2_intermediate_entirecomplexplane}.

    Now assume \(f\) is not necessarily of finite order and \(R_0=\infty\). Then by \cref{prop:nevanlinnalogdiffproximityestimate} with \(R=r+\tfrac1{T(r)}\),
    \begin{gather*}
        m\qty(r,\frac{f'}f)<\order{\logp T\qty(r+\frac1{T(r)},f)}+\order{1}+\order{\logp r}\\
        m\qty(r,\frac{\phi'}\phi)<\order{\logp T\qty(r+\frac1{T(r)},f)}+\order{1}+\order{\logp r}.
    \end{gather*} By \cref{lem:nevanlinnaborel}, there is some exceptional set \(E_0\) of linear measure not exceeding 2 such that \[\logp T\qty(r+\frac1{T(r)},f)\leq\logp T\qty(r+\frac1{T(r)},f)+\order{1},\]
    thus for \(r\to\infty\) with \(r\notin E_0\),
    \[S(r,f)=\order{\logp T\qty(r,f)}+\order{\logp r}.\]
    Lastly, assume that \(R_0<\infty\). Now apply \cref{prop:nevanlinnalogdiffproximityestimate} with \(R=r+\frac{R_0-r}{\ee T(r,f)}\) outside an exceptional set satisfying \(\textstyle\int_E\frac{\dd{r}}{R_0-r}<\infty\) to get that
    \begin{gather*}
        m\qty(r,\frac{f'}f)<\order{\logp T(R,f)}+\order{1}+\order{\logp\qty(r+\frac{R_0-r}{\ee T(r,f)})}+\order{\logp\frac{\ee T(r,f)}{R_0-r}},\\
        S(r,f)=\order{\logp T(r,f)+\log\frac1{R_0-r}}.
    \end{gather*}
    The final assertion is also evident from \cref{lem:nevanlinnaborel}.
\end{proof}
\begin{proposition}\label{prop:nevanlinnaorderrationalcondition}
    If \(f:\mathbb{C}\to\extcomplex\) is meromorphic and \(T(r,f)=\order{\log r}\), then \(f\) is a rational function.
\end{proposition}
\begin{proof}
    We must have (for any \(a\in\extcomplex\)) \(N(r,a,f)=\order{\log r}\). We assert that \(f\) attains each value finitely many times. Otherwise, for each \(k\in\mathbb{N}\), there exists \(r'\) such that \(\forall r>r'\), \(n\qty(r,a,f)>k\). Then \[N(r,a,f)=c+\int_{r'}^{r}\frac{n(x,a,f)-n(0,a,f)}{x}\ddx+n(0,a,f)\log r>c+k\log r.\]
    Thus, \(N\) eventually grows faster than \(k\log r\) for any \(k\). It follows that each \(a\) is attained finitely many times by contradiction. Construct \(g=\order{\abs{z}^d}\) to be a polynomial with its zeros at each pole (finitely many) of \(f\).

    Then \(h=fg\) is entire, and furthermore, \(T(r,h)\leq T(r,f)+T(r,g)\). Then \(T(r,f)=\order{\log r}\) and \[T(r,g)=m(r,g)<\logp M(r,g)=\order{\log r}.\]
    Then \[\log M(r,h)\leq 3T(2r,h)=\order{\log r}\] by \cref{thm:nevanlinnaentirefunctionmaximummodulussandwich}. Then \[M(r,h)\leq ar^b,\] where \(a\), \(b\) are constants. Thus the growth of \(h\) may not exceed that of a polynomial. By Cauchy's Estimate (\cref{thm:cauchysestimate}), for any natural number \(n\) and any \(r>0\), \[M\qty(r,h^{(n)})\leq n!r^{b-n}.\]
    For \(n>b\), let \(r\to\infty\) to find that the derivatives of \(h\) eventually vanish. Therefore, \(h\) is a polynomial, and consequently \(f\) is the quotient of two polynomials.
\end{proof}
\begin{definition}
    Let \(f:D\qty(0,R_0)\to\extcomplex\) be meromorphic for \(R_0\in\mathbb{R}_{>0}\cup\qty{\infty}\). Then \(f\) is said to be \textit{admissible} if either of the following are true:
    \begin{enumerate}
        \item \(R_0=\infty\) and \(f\) is non-constant.
        \item \(R_0<\infty\) and \(\varlimsup_{r\to R_0^-}\tfrac{T(r,f)}{\log\qty[\flatfrac{1}{\qty(R_0-r)}]}=\infty\).
    \end{enumerate}
\end{definition}
These definitions ensure that \(S(r,f)\) is generally sufficiently small:
\begin{theorem}
    An admissible meromorphic function \(f:D\qty(0,R_0)\to\extcomplex\) (given for \(R_0\in\mathbb{R}_{>0}\cup\qty{\infty}\)) pertains to one of the following cases:
    \begin{enumerate}
        \item If \(R_0=\infty\) and \(f\) has finite order, then \(S(r,f)=o\qty[T(r,f)]\) for \(r\to R_0^-\) through all values.
        \item If \(R_0=\infty\) and \(f\) has infinite order, then \(S(r,f)=o\qty[T(r,f)]\) for \(r\to R_0^-\) outside a set \(E_0\) of finite linear measure, which depends on \(f\) but not on \(\qty{a_\nu}\) and \(q\).
        \item If \(R_0<\infty\), then \(S(r,f)=o\qty[T(r,f)]\) for \(r\to R_0^-\) through a sequence \(\qty{r_n}_n\) dependent on \(f\) but not on \(\qty{a_\nu}\) and \(q\).
    \end{enumerate}
\end{theorem}
\begin{proof}
    Assume the conditions of the theorem.
    \begin{enumerate}
        \item Assume \(R_0=\infty\) and \(f\) has finite order. Then \(T(r,f)=\order{r^{\rho+\varepsilon}}\), \(r^{\rho-\varepsilon}=o\qty[T(r,f)]\) (assuming that \(\rho>0\)), so \(\log r=o\qty[T(r,f)]\). Therefore, by \cref{thm:nevanlinnasecondfundamentaltheorempart2}, it follows that through all values as \(r\to\infty\),
            \[S(r,f)=\order{\log T(r,f)}+\order{\log r}=\order{\log T(r,f)}=o\qty[T(r,f)].\]
            If \(\rho=0\), we have for any \(\varepsilon\) that for sufficiently large \(r\), either \(\log r=o\qty(T(r,f))\) or \(T\qty(r,f)=\order{\log r}\). The former case is the same as above.

            In the latter case, \cref{prop:nevanlinnaorderrationalcondition} gives that \(f\) is rational. Hence, as \(r\to\infty\), \(M\qty(r,\tfrac{f'}{f}),M\qty(r,\sum_\nu\tfrac{f'}{f-a_\nu})\to 0\). Hence for sufficiently large \(r\), \(m\qty(r,\tfrac{f'}f)\) and \(m\qty(r,\sum_\nu\tfrac{f'}{f-a_\nu})\) vanish and \(S(r,f)=\order{1}\), \(\therefore\flatfrac{S}{T}\to 0\).
        \item Assume \(R_0=\infty\) and \(f\) has infinite order. Then by \cref{thm:nevanlinnasecondfundamentaltheorempart2}, it follows that for \(r\to R_0^-\) outside a set \(E_0\) of finite linear measure, \(S(r,f)=\order{\log T(r,f)}+\order{\log r}\). Since \(\tfrac{\log T(r,f)}{\log r}\to\infty\), then \(\log r=o\qty[T(r,f)]\), and thus \(S(r,f)=o\qty[T(r,f)]\) outside this exceptional set.
        \item Assume \(R_0<\infty\). Construct two sequences \(\qty{\rho}_n,\qty{\rho'}_n\) such that \(\rho_n<\rho'_n\), \(\rho_n,\rho'_n\to R_0^-\), and \[\lim_{n\to\infty}\frac{T\qty(\rho_n,f)}{\log\qty[\flatfrac1{\qty(R_0-\rho_n)}]}=\infty\] (observe that this is no longer a limit superior) and \(\rho'_n=R_0-\ee^{-2}\tfrac{R_0-\rho_n}2\), so that \(R_0-\rho'_n<\ee^{-2}\qty(R_0-\rho_n)\).

            Then by \cref{thm:nevanlinnasecondfundamentaltheorempart2}, there exists some \(r_n\in(\rho_n,\rho'_n)\) such that \(r_n\notin E_0\). It follows that \(R_0-\rho_n>R_0-r_n>R_0-\rho'_n=\tfrac{R_0-\rho_n}{2\ee^2}\). Then for \(n\to\infty\), \[\frac1{R_0-\rho_n}=o\qty(T\qty(\rho_n))=o\qty(T\qty(r_n)),\] since \(T\) is a non-decreasing function of radius. Then \[S\qty(r_n,f)=\order{\logp T(r_n,f)+\log\qty[o\qty(T\qty(r_n,f))]}=o\qty[T(r_n,f)]\qq{as}n\to\infty.\qedhere\]
    \end{enumerate}
\end{proof}
