\begin{theorem}[Pompeiu's Theorem]\label{theorem:pompeiu}
    Let \(U\subseteq\mathbb{C}\) where $\partial U$ is piecewise smooth. Let \(f(z)\in C^1(\bar{U})\), where $\bar{U}$ represents the closure of \(U\). Then $\forall z\in U$, \begin{equation}
        f(z)=\frac{1}{2\pi i}\left(\int_{\partial U}\frac{f(\zeta)}{\zeta-z}d\zeta-\int_{U}\frac{\partial f(\zeta)}{\partial\bar{\zeta}}\frac{d\bar{\zeta}\wedge d\zeta}{\zeta-z}\right).
    \end{equation}
\end{theorem}
